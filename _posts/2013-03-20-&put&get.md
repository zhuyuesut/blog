```c++
#include <iostream>
using namespace std;

int &put(int);
int get(int);

int vals[10];
int error=-1;

int main(int argc, const char * argv[])
{
    cin>>put(0);
    cin>>put(1);
    cin>>put(2);
    
    cout<<get(0)<<endl;
    cout<<get(1)<<endl;
    cout<<get(2)<<endl;
    
    cin>>put(12);
    
    return 0;
}

int &put(int n)
{
    if (n>=0&&n<10)
        return vals[n];
    
    cout<<"error:"<<error<<endl;
    
    int *p=NULL;
    return *p;
}

int get(int n) {
    if (n>=0&&n<10)
        return vals[n];
    
    cout<<"error:"<<error<<endl;
    
    return error;
}
```
