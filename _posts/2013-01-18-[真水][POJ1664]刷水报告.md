```c
#include<stdio.h>

int main()
{
    int m, n, k[11][11], t;
    k[0][0] = 1;
    for (m = 0; m < 11; m++)
    {
        k[m][1] = 1;
        for (n = 2; n <= m; n++)
            k[m][n] = k[m][n - 1] + k[m - n][n];
        while (n < 11)
            k[m][n++] = k[m][m];
    }
    scanf("%d", &t);
    while (t--)
    {
        scanf("%d %d", &m, &n);
        printf("%d\n", k[m][n]);
    }
    return 0;
}
```

当n>m时，f(m，n)=f(m,m)(其它的是空盘子)，当n=1时f(m，n)=1(只有一个盘子)，否则f(m，n)=f(m-1，n)+f(m-n，n)(所有盘子都有和有空的)
