#include<stdio.h>
#define lson l,m,root<<1
#define rson m+1,r,root<<1|1
const int maxn=50010;
int sum[maxn<<2];
void push_up(int root)
{
	sum[root]=sum[root<<1]+sum[root<<1|1];
}
void build(int l,int r,int root)
{
	int m;
	if(l==r)
	{
		scanf("%d",sum+root);
		return;
	}
	m=(l+r)>>1;
	build(lson);
	build(rson);
	push_up(root);
}
void update(int x,int add,int l,int r,int root)
{
	int m;
	if(l==r)
	{
		sum[root]+=add;
		return ;
	}
	m=(l+r)>>1;
	if(x<=m)
	{
		update(x,add,lson);
	}
	if(m+1<=x)
	{
		update(x,add,rson);
	}
	push_up(root);
}
int query(int L,int R,int l,int r,int root)
{
	int m,result=0;
	if(L<=l&&r<=R)
	{
		return sum[root];
	}
	m=(l+r)>>1;
	if(L<=m)
	{
		result+=query(L,R,lson);
	}
	if(m+1<=R)
	{
		result+=query(L,R,rson);
	}
	return result;
}
int main()
{
	int t,cases,n,a,b;
	char s[8];
	scanf("%d",&t);
	for(cases=1;cases<=t;cases++)
	{
		printf("Case %d:\n",cases);
		scanf("%d",&n);
		build(1,n,1);
		while(scanf("%s",s),s[0]-'E')
		{
			scanf("%d%d",&a,&b);
			if(s[0]=='Q')
			{
				printf("%d\n",query(a,b,1,n,1));
				continue;
			}
			if(s[0]=='S')
			{
				b=-b;
			}
			update(a,b,1,n,1);
		}
	}
	return 0;
}
