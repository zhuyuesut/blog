```c++

#include<stdio.h> 
#define N 1000100

int prime_hash[N] = {1,1}, prime[N];
int prime_num = 0;

void make_prime()
{
    int i, j, step=2;
    for(i=5; i<N; i+=step, step=6-step)
    {
        if(!prime_hash[i])
        {
            prime[prime_num++] = i;
        }
        for(j=0; j<prime_num && i*prime[j]<N; j++)
        {
            prime_hash[i*prime[j]] = 1;
            if(!(i%prime[j]))
            {
                break;
            }
        }
    }
}

int is_prime(int x)
{
    if(x==2 || x==3)
    {
        return 1;
    }
    else
    {
        if(x%2 && x%3 && prime_hash[x]==0)
        {
            return 1;
        }
        else
        {
            return 0;
        }
    }
}
```
