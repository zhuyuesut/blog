#include<stdio.h>
int main()
{
    int n, i, j, a, dp[128] = { 0 };
    scanf("%d", &n);
    i = n;
    while (i--)
    {
        for (j = i; j < n; j++)
        {
            scanf("%d", &a);
            dp[j] = (dp[j] > dp[j + 1]) ? dp[j] + a : dp[j + 1] + a;
        }
    }
    for (i = 1, j = 0; i <= n; i++)
        if (dp[j] < dp[i])
            j = i;
    printf("%d\n", dp[j]);
    return 0;
}
